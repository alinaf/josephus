Josephus problem


One hundred people are standing in a circle, numbered 1-100. Person #1 has a sword. He kills the next person (i.e. Person #2) and gives the sword to the next surviving person (i.e. Person #3). This goes on until only one person remains. Who survives at the end?

Number of people:

Answer: Person #73

Code


function josephus(n) {
  const n1 = {value: 1, next: null};
  let temp = n1;

  for(i = 2; i <= n; i++) {
    temp.next = {value: i, next: null};
    temp = temp.next;
  }

  numElements = n;
  temp.next = n1; // circular
  temp = n1; // start at person 1

  while(numElements > 1) {
    temp.next = temp.next.next;
    numElements--;
    temp = temp.next;
  }

  console.log(temp.value);
}
